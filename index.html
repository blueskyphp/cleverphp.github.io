2.design patterns(observer)


            /***
                it likes a watcher,when what watched changes,do something then(mostly notify others)...
            ***/
            var watcher = (function(){
                var require_watches = [];
                var put_el = function(v){
                    require_watches.push(v);
                };
                var doSome = function(){
                    console.log('changed');
                };
                return {
                    list : require_watches,
                    num : 0,
                    changed : false,
                    putEl : put_el,
                    doSome : function(){
                        doSome();
                        this.changed = false;
                    },
                    listenStart : function(){
                        setInterval(this.listen,1000);
                    },
                    listen : function(){
                        var len = this.list.length;
                        if(len != num){
                            this.changed = true;
                            this.doSome();
                            this.num = len;
                        }
                    }
                };
            })();
            watcher.listenStart();//when putEl,will invoke watching func

JAVA

Anyone who is programmer should know it...

1.hash examples


        import java.security.*;
        public class Hash {
            /**
             * @param txt, text in plain format
             * @param hashType MD5 OR SHA1
             * @return hash in hashType 
             */
            public static String getHash(String txt, String hashType) {
                try {
                        java.security.MessageDigest md = java.security.MessageDigest.getInstance(hashType);
                        byte[] array = md.digest(txt.getBytes());
                        StringBuffer sb = new StringBuffer();
                        for (int i = 0; i < array.length; ++i) {
                            sb.append(Integer.toHexString((array[i] & 0xFF) | 0x100).substring(1,3));
                        }
                        return sb.toString();
                    } catch (java.security.NoSuchAlgorithmException e) {
                        //error action
                    }
                    return null;
            }
            public static String md5(String txt) {
                return Hash.getHash(txt, "MD5");
            }

            public static String sha1(String txt) {
                return Hash.getHash(txt, "SHA1");
            }
        }

3.sql like used incr/incrby,decr/decrby


            $redis = new Redis();
            $redis->connect("127.0.0.1","6379");
            $redis->set("views",1);
            $redis->incr("views");
            echo $redis->get("views"); //2
            $redis->decr("views");
            echo $redis->get("views"); //1
            $redis->incrby("views",100);
            echo $redis->get("views"); //101
            $redis->decrby("views",100);
            echo $redis->get("views"); //1
            //ps:if you want to delete a redis key,just use del method,examples:$redis->del("views");
            //if you want to get all keys,use $redis->keys("*");
            //if just for special patterns,use $redis->keys("pattern_*") will match all keys that prefix is pattern_;
